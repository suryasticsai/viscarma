Phase I Prototype — Complete Documentation

1. Overview

VisCarma is a self‑hosted, offline, multi‑modal AI agent that uses Playwright to inspect a running web application, a local large language model (LLM) to reason about code and UI, and the GitHub API to commit changes and open pull requests. In Phase I, it accepts a natural language prompt (bug fix or feature request), captures the app’s live state, consults the offline LLM, applies the suggested file‑level edits, and creates a PR — all without any external cloud dependency.

3. Prerequisites

  • Node.js ≥ 18
  • Ollama installed and running locally
  • Playwright browsers installed (npx playwright install --with-deps chromium)
  • Git installed and a local clone of the target repository
  • A GitHub Personal Access Token with repo scope (set as GITHUB_TOKEN env variable)
  • The model qwen2:7b-vl pulled via Ollama: ollama pull qwen2:7b-vl

6. Core Modules

6.1 lib/capture.js — Playwright Reconnaissance

const { chromium } = require('playwright');

async function captureAppState(appUrl) {
  const browser = await chromium.launch({ headless: true });
  const context = await browser.newContext();
  const page = await context.newPage();

  const consoleErrors = [];
  page.on('console', msg => {
    if (msg.type() === 'error') {
      consoleErrors.push({ text: msg.text(), location: msg.location() });
    }
  });

  await page.goto(appUrl, { waitUntil: 'networkidle' });

  const screenshotBuffer = await page.screenshot({ fullPage: true, type: 'png' });
  const dom = await page.content();
  await browser.close();

  return {
    screenshotBase64: screenshotBuffer.toString('base64'),
    dom,
    consoleErrors
  };
}

module.exports = { captureAppState };

6.2 lib/model.js — Ollama Oracle

const { default: ollama } = require('ollama');

async function queryModel(model, ollamaUrl, prompt, context) {
  const { screenshotBase64, dom, consoleErrors, sourceFiles } = context;
  const content = `
Mission: ${prompt}

Console errors: ${JSON.stringify(consoleErrors)}
Source files (JSON): ${JSON.stringify(sourceFiles, null, 2)}
DOM (first 5000 chars): ${dom.substring(0, 5000)}

Return ONLY a JSON array: [{ "file": "...", "oldLine": "...", "newLine": "..." }]`;

  const response = await ollama.chat({
    model,
    messages: [{ role: 'user', content, images: [screenshotBase64] }],
    format: 'json',
    options: { temperature: 0 }
  });

  return JSON.parse(response.message.content);
}

module.exports = { queryModel };

6.3 lib/git.js — Divine Pull Requests

const { Octokit } = require('@octokit/rest');
const fs = require('fs');
const path = require('path');

async function applyChangesAndOpenPR(config, changes, missionTitle) {
  const octokit = new Octokit({ auth: process.env.GITHUB_TOKEN });
  const { owner, repo, baseBranch } = config.github;
  const branchName = `visCarma/${Date.now()}`;

  // Create branch from base
  const { data: baseRef } = await octokit.git.getRef({
    owner, repo, ref: `heads/${baseBranch}`
  });
  await octokit.git.createRef({
    owner, repo, ref: `refs/heads/${branchName}`, sha: baseRef.object.sha
  });

  for (const change of changes) {
    const localPath = path.join(config.repoPath, change.file);
    let content = fs.readFileSync(localPath, 'utf8');
    content = content.replace(change.oldLine, change.newLine);
    fs.writeFileSync(localPath, content);

    await octokit.repos.createOrUpdateFileContents({
      owner, repo, path: change.file,
      message: `VisCarma: ${missionTitle}`,
      content: Buffer.from(content).toString('base64'),
      branch: branchName
    });
  }

  const pr = await octokit.pulls.create({
    owner, repo,
    title: `VisCarma: ${missionTitle}`,
    head: branchName, base: baseBranch,
    body: `Auto‑generated by VisCarma.\n\nMission: ${missionTitle}`
  });
  console.log(`✅ PR created: ${pr.data.html_url}`);
  return pr.data.html_url;
}

module.exports = { applyChangesAndOpenPR };

6.4 agent.js — Master Orchestrator

const fs = require('fs');
const path = require('path');
const { captureAppState } = require('./lib/capture');
const { queryModel } = require('./lib/model');
const { applyChangesAndOpenPR } = require('./lib/git');
const CONFIG = require('./config/visCarma.json');

function getSourceFiles(repoPath) {
  const files = {};
  function walk(dir) {
    for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
      const full = path.join(dir, entry.name);
      if (entry.isDirectory()) {
        if (['node_modules','.git'].includes(entry.name)) continue;
        walk(full);
      } else if (entry.name.match(/\.(js|html|css|json|ts|jsx|tsx)$/)) {
        files[path.relative(repoPath, full)] = fs.readFileSync(full, 'utf8');
      }
    }
  }
  walk(repoPath);
  return files;
}

async function runMission(prompt) {
  console.log(`🕵️ VisCarma embarking on mission: "${prompt}"`);
  const { screenshotBase64, dom, consoleErrors } = await captureAppState(CONFIG.appUrl);
  const sourceFiles = getSourceFiles(CONFIG.repoPath);
  const changes = await queryModel(CONFIG.model, CONFIG.ollamaUrl, prompt, {
    screenshotBase64, dom, consoleErrors, sourceFiles
  });

  if (!changes.length) return console.log('No changes required.');
  await applyChangesAndOpenPR(CONFIG, changes, prompt);
  console.log('🎯 Mission complete.');
}

if (require.main === module) {
  let mission = process.argv[2] || 
    fs.readFileSync(path.join(__dirname, 'prompts', 'mission.txt'), 'utf8').trim();
  runMission(mission).catch(err => { console.error(err); process.exit(1); });
}

module.exports = { runMission };

7. Running the Prototype

Setup: Clone your target app, start its dev server at the configured appUrl. Install VisCarma dependencies:

cd visCarma
npm init -y
npm install playwright @octokit/rest ollama
npx playwright install --with-deps chromium
export GITHUB_TOKEN=ghp_xxxxxxxxxxxxx

Execute a mission:

node agent.js "Add a dark mode toggle button"

Or prepare prompts/mission.txt and run without arguments.

8. Logging & Monitoring

All console output is logged. Redirect to file for persistent logs:

node agent.js "Fix login bug" >> logs/mission_$(date +%s).log 2>&1

9. Known Limitations (Phase I)

  • File update SHA – Updating existing files on a branch requires the file’s SHA; the prototype assumes a clean branch. Production version will fetch SHA first.
  • Single‑page capture – Only one URL is observed. Multi‑page apps need route crawling.
  • LLM output reliability – Occasionally invalid JSON; retry logic recommended.
  • No self‑verification – The PR code is not automatically tested (planned for Phase II).
  • Large codebases – All source files are sent; filter or scope based on diff for efficiency.

10. Next Steps (Phase II Roadmap)

  • Validation loop – Run Playwright tests on the generated PR branch; if tests fail, feed error back to LLM for re‑fix.
  • Scheduled/time‑bounded missions – Cron/systemd timers for surveillance windows (e.g., only during business hours).
  • Automatic bug detection trigger – Continuous mode: console error or visual regression → instant PR.
  • Diff‑based edits – Use proper patch libraries instead of simple string replacement.
  • GPG‑signed commits & branch protection – Authenticate VisCarma’s commits with a bot GPG key.
